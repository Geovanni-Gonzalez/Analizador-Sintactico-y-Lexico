<tipo> ::= "int" | "float" | "bool" | "char" | "string"

<tipoArreglo> ::= "int" | "float"

<entero> ::= "-"? <enteroPos>

<enteroPos> ::= [1-9][0-9]* | "0"

<flotante> ::= <entero> "." [0-9]+

<fraccion> ::= <entero> "/" <enteroPos>

<exponente> ::= <entero> "e" <enteroPos>

<booleano> ::= "true" | "false"

<char> ::= "'" ([a-zA-Z0-9_] ) "'"

<string> ::= '"' ([a-zA-Z0-9_])* '"'

<id> ::= [a-zA-Z_][a-zA-Z0-9_]*

<idTail> ::= "<<" <expr> "," <expr> ">>" | "<|" <args> "|>" |

<programa> ::= <funciones_opt> <main> <funciones_opt>

<funciones_opt> ::= <funciones> | ε

<funciones> ::= <funcion> <funciones> | <funcion>

<main> ::= "empty" "~" "__main__" "<|" "|>" <bloque>

<funcion> ::= <tipoRetorno> "~" <id> "<|" <parametros> "|>" <bloque>

<tipoRetorno> ::= "int" | "float" | "bool" | "char"

<parametros> ::= <lista_params> | ε

<lista_params> ::= <param> ("," <param>)*

<param> ::= <tipo> "~" <id>

<bloque> ::= "|:" <listaSentencias> ":|"

<listaSentencias> ::= <item>+

<item> ::= <sentenciaSimple> "!" | <control>

<sentenciaSimple> ::= <decl> | <asignacion> | <expr> | <io> | <retorno> | <break>

<decl> ::= <tipo> "~" <id> <declTail>

<declTail> ::= "<-" <expr> | "|:" <enteroPos> "," <enteroPos> ":|" <declArrTail> |

<declArrTail> ::= "<-" <filasArreglo> |

<filasArreglo> ::= <filaArreglo>+

<filaArreglo> ::= <filaEnteros> | <filaFlotantes>

<filaEnteros> ::= "|:" <entero> ("," <entero>)* ":|"

<filaFlotantes> ::= "|:" <flotante> ("," <flotante>)* ":|"

<asignacion> ::= <id> <asigTail>

<asigTail> ::=  "<-" <expr> | "<<" <expr> "," <expr> ">>" "<-" <expr>

<expr> ::= <exprLog>

<exprLog> ::= <exprRel> ( <opLog> <exprRel> )*

<opLog> ::= "@" | "#"

<exprRel> ::= <opRel> "<|" <expr> "," <expr> "|>" | <exprArit>

<opRel> ::= "greather_t" | "less_t" | "greather_te" | "less_te" | "equal" | "n_equal"

<exprArit> ::= <termino> ( ("+" | "-") <termino> )*

<termino> ::= <potencia> ( ("*" | "/" | "%") <potencia> )*

<potencia> ::= <factor> "^" <potencia> | <factor>

<factor> ::= "-" <factor> | "++" <id> | "--" <id> | "$" <id> | <base>

<base> ::= <literal> | <id> <idTail> | "<|" <expr> "|>"

<literal> ::= <entero> | <flotante> | <booleano> | <char> | <string>

<args> ::= <lista_args> | ε

<lista_args> ::= <expr> ("," <expr>)*

<control> ::= <ifElse> | <doWhile> | <switch>

<ifElse> ::= "if" "<|" <expr> "|>" <bloque> | "if" "<|" <expr> "|>" <bloque> "else" <bloque>

<doWhile> ::= "do" <bloque> "while" "<|" <expr> "|>"

<switch> ::= "switch" "<|" <expr> "|>" "|:" <casos> ":|"

<casos> ::= <caso>+

<caso> ::= "case" <entero> "~" <bloque> | "default" "~" <bloque>

<io> ::= <lectura> | <escritura>

<lectura> ::= "cin" "~" <id>

<escritura> ::= "cout" "~" <salida>

<salida> ::= <expr> ("," <expr>)*

<retorno> ::= "return" "~" <expr>

<break> ::= "break"
